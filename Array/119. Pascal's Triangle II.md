      ```
      // O(K^2)time
      // O(K)space
      public List<Integer> getRow(int rowIndex) {
          List<Integer> pre = new LinkedList<>();
          List<Integer> res = new LinkedList<>();
          List<Integer> temp;
          while (rowIndex >= 0) {
              rowIndex--;
              res.add(1);
              if (pre.size() > 0) {
                  while (pre.size() > 1) {
                      res.add(pre.remove(0) + pre.get(0));
                  }
                  res.add(pre.remove(0));
              }
              temp = res;
              res = pre;
              pre = temp;
          }
          return pre;
      }
      ```
