      //O(N*N)time O(N*N)space
      public int regionsBySlashes(String[] grid) {
          int count = 0;
          int r = grid.length;
          int c = grid[0].length();
          int[][] record = new int[r * 3][c * 3];
          for (int i = 0; i < r; i++) {
              for (int j = 0; j < c; j++) {
                  if (grid[i].charAt(j) == '/') {
                      record[i * 3][j * 3 + 2]
                              = record[i * 3 + 1][j * 3 + 1]
                              = record[i * 3 + 2][j * 3]
                              = 1;
                  } else if (grid[i].charAt(j) == '\\') {
                      record[i * 3][j * 3]
                              = record[i * 3 + 1][j * 3 + 1]
                              = record[i * 3 + 2][j * 3 + 2]
                              = 1;
                  }
              }
          }
          for (int i = 0; i < record.length; i++) {
              for (int j = 0; j < record[i].length; j++) {
                  if (record[i][j] == 0) {
                      dfs(i, j, record);
                      count++;
                  }
              }
          }
          return count;
      }

      public void dfs(int i, int j, int[][] record) {
          if (i >= 0 && i < record.length
                  && j >= 0 && j < record[i].length
                  && record[i][j] == 0) {
              record[i][j] = 1;
              dfs(i, j - 1, record);
              dfs(i, j + 1, record);
              dfs(i - 1, j, record);
              dfs(i + 1, j, record);
          }
      }

      ```
      public int regionsBySlashes(String[] grid) {
          int r = grid.length;
          int c = grid[0].length();
          int count = r * c * 4;
          UF uf = new UF(count);
          for(int i = 0; i < r; i++){
              for(int j = 0; j < c; j++){
                  char t = grid[i].charAt(j);
                  int leftTop = i * 2 * c * 2 + j * 2;
                  if(t == '/'){
                      uf.remove(leftTop + 1);
                      uf.remove(leftTop + 2 * c);
                      if(j + 1 < c){
                          char next = grid[i].charAt(j + 1);
                          if(next == '\\'){
                              uf.union(leftTop + 2 * c + 1, leftTop + 2 * c + 2);
                          }else{
                              uf.union(leftTop + 2 * c + 1, leftTop + 2);
                          }
                      }
                      if(i + 1 < r){
                          char next = grid[i + 1].charAt(j);
                          if(next == '\\'){
                              uf.union(leftTop + 2 * c + 1, leftTop + 4 * c + 1);
                          }else{
                              uf.union(leftTop + 2 * c + 1, leftTop + 4 * c);
                          }
                      }
                  }else if(t == '\\'){
                      uf.remove(leftTop);
                      uf.remove(leftTop + 2 * c + 1);

                      if(j + 1 < c){
                          char next = grid[i].charAt(j + 1);
                          if(next == '\\'){
                              uf.union(leftTop + 1, leftTop + 1 + 2 * c + 1);
                          }else{
                              uf.union(leftTop + 1, leftTop + 1 + 1);
                          }
                      }
                      if(i + 1 < r){
                          char next = grid[i + 1].charAt(j);
                          if(next == '\\'){
                              uf.union(leftTop + 2 * c, leftTop + 4 * c + 1);
                          }else{
                              uf.union(leftTop + 2 * c, leftTop + 4 * c);
                          }
                      }
                  }else{
                      uf.union(leftTop, leftTop + 1);
                      uf.union(leftTop + 2 * c, leftTop + 2 * c + 1);
                      uf.union(leftTop, leftTop + 2 * c);

                      if(j + 1 < c){
                          char next = grid[i].charAt(j + 1);
                          if(next == '\\'){
                              uf.union(leftTop + 1 + 2 * c, leftTop + 1 + 2 * c + 1);
                          }else {
                              uf.union(leftTop + 1, leftTop + 1 + 1);
                          }
                      }
                      if(i + 1 < r){
                          char next = grid[i + 1].charAt(j);
                          if(next == '/'){
                              uf.union(leftTop + 2 * c, leftTop + 4 * c);
                          }else{
                              uf.union(leftTop + 2 * c + 1, leftTop + 4 * c + 1);
                          }
                      }
                  }
              }
          }
          return uf.getCount();

      }

      class UF{
          int count;
          int[] arr;
          public UF(int n){
              count = n;
              arr = new int[n];
              for(int i = 0; i < n; i++){
                  arr[i] = i;
              }
          }

          void remove(int a){
              if(arr[a] == -1) return;
              arr[a] = -1;
              count--;
          }

          int find(int a){
              if(a != arr[a]){
                  arr[a] = find(arr[a]);
              }
              return arr[a];
          }

          boolean union(int a, int b){
              if(find(a) != find(b)){
                  arr[find(a)] = find(b);
                  count--;
                  return true;
              }
              return false;
          }

          int getCount(){
              return count;
          }
      }
      ```
